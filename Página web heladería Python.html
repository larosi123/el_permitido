🧠 3. Actualizar app.py para incluir pagos

Agregamos una nueva ruta /pago para procesar el pago con Mercado Pago y una opción para pagar en efectivo.

python
import mercadopago
from flask import Flask, render_template, request, redirect, url_for

app = Flask(__name__)

sdk = mercadopago.SDK("TU_ACCESS_TOKEN_DE_PRUEBA")

GUSTOS = [
    "Chocolate", "Chocolate Blanco", "Chocolate Amargo", "Dulce de Leche",
    "Dulce de Leche Granizado", "Vainilla", "Frutilla", "Banana Split",
    "Tramontana", "Crema Oreo", "Pistacho"
]

@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        tamaño = request.form['tamaño']
        gustos = request.form.getlist('gustos')
        metodo_pago = request.form['metodo_pago']
        if len(gustos) > 3:
            error = "¡Solo puedes elegir hasta 3 gustos por pote!"
            return render_template('index.html', gustos=GUSTOS, error=error)
        
        if metodo_pago == 'mercadopago':
            preference_data = {
                "items": [
                    {
                        "title": f"Pote de {tamaño}",
                        "quantity": 1,
                        "currency_id": "ARS",
                        "unit_price": calcular_precio(tamaño)
                    }
                ],
                "back_urls": {
                    "success": url_for('confirmacion', _external=True),
                    "failure": url_for('index', _external=True)
                },
                "auto_return": "approved"
            }
            preference_response = sdk.preference().create(preference_data)
            preference = preference_response["response"]
            return redirect(preference["init_point"])
        else:
            return render_template('confirmacion.html', tamaño=tamaño, gustos=gustos, metodo_pago=metodo_pago)

    return render_template('index.html', gustos=GUSTOS)

@app.route('/confirmacion')
def confirmacion():
    tamaño = request.args.get('tamaño', 'N/A')
    gustos = request.args.getlist('gustos')
    metodo_pago = request.args.get('metodo_pago', 'mercadopago')
    return render_template('confirmacion.html', tamaño=tamaño, gustos=gustos, metodo_pago=metodo_pago)

@app.route('/promociones')
def promociones():
    return "<h1>Promociones</h1><p>¡2x1 todos los martes!</p>"

def calcular_precio(tamaño):
    precios = {
        "1 kilo": 3000,
        "3/4 kilo": 2500,
        "1/2 kilo": 2000,
        "1/4 kilo": 1200
    }
    return precios.get(tamaño, 0)

if __name__ == '__main__':
    app.run(debug=True)
